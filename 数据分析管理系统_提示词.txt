3.2 智能主键检测器（app/utils/key_detector.py）

// python
import pandas as pd
from typing import List, Optional, Tuple
import re

class KeyDetector:
    """智能主键检测器"""

    # 常见主键关键词
    KEYWORD_PATTERNS = [
        r'id$', r'_id$', r'编号', r'代码', r'名称', r'number$',
        r'code$', r'用例id', r'用例编号', r'test.*id', r'item.*id'
    ]

    def __init__(self, df1: pd.DataFrame, df2: pd.DataFrame):
        """
        初始化主键检测器

        Args:
            df1: 文件1的DataFrame
            df2: 文件2的DataFrame
        """
        self.df1 = df1
        self.df2 = df2
        self.common_columns = list(set(df1.columns) & set(df2.columns))

    def detect_key_columns(self, max_keys: int = 2) -> List[str]:
        """
        自动检测主键列

        Args:
            max_keys: 最多返回多少个主键列

        Returns:
            主键列名列表
        """
        if not self.common_columns:
            return []

        # 1. 基于关键词匹配
        keyword_matches = []
        for col in self.common_columns:
            col_lower = col.lower()
            for pattern in self.KEYWORD_PATTERNS:
                if re.search(pattern, col_lower):
                    keyword_matches.append(col)
                    break

        if keyword_matches:
            return keyword_matches[:max_keys]

        # 2. 基于数据特征检测
        # 选择唯一值比例高且数据类型合适的列
        candidates = []
        for col in self.common_columns:
            unique_ratio = len(self.df1[col].dropna().unique()) / max(1, len(self.df1[col].dropna()))

            # 唯一值比例高（>0.5）且数据类型不是纯数值
            if unique_ratio > 0.5:
                dtype = str(self.df1[col].dtype)
                is_numeric = dtype in ['int64', 'float64', 'int32', 'float32']

                # 文本类型或ID类型的优先级更高
                score = unique_ratio
                if 'str' in dtype or 'object' in dtype:
                    score += 0.3
                elif is_numeric:
                    score += 0.1

                candidates.append((col, score))

        # 按得分排序
        candidates.sort(key=lambda x: x[1], reverse=True)

        return [col for col, _ in candidates[:max_keys]]

    def build_key_index(self, df: pd.DataFrame, key_columns: List[str]) -> dict:
        """
        构建主键索引

        Args:
            df: DataFrame
            key_columns: 主键列名列表

        Returns:
            {tuple(主键值): 行索引}
        """
        if not key_columns:
            return {}

        # 检查主键列是否存在
        for col in key_columns:
            if col not in df.columns:
                raise ValueError(f"主键列 '{col}' 不存在")

        # 构建索引
        key_index = {}
        for idx, row in df.iterrows():
            # 处理空值
            key_values = []
            for col in key_columns:
                val = row.get(col)
                if pd.isna(val):
                    val = None
                else:
                    val = str(val).strip()
                key_values.append(val)

            # 转为元组作为字典键
            key_tuple = tuple(key_values)

            # 如果主键重复，保留第一个
            if key_tuple not in key_index:
                key_index[key_tuple] = idx

        return key_index

    def get_sample_keys(self, df: pd.DataFrame, key_columns: List[str], sample_size: int = 5) -> List[str]:
        """
        获取主键样本值

        Args:
            df: DataFrame
            key_columns: 主键列名列表
            sample_size: 样本数量

        Returns:
            主键样本值列表（字符串形式）
        """
        if not key_columns:
            return []

        samples = []
        for idx, row in df.head(sample_size).iterrows():
            key_parts = []
            for col in key_columns:
                val = row.get(col)
                if pd.isna(val):
                    key_parts.append('None')
                else:
                    key_parts.append(str(val))
            samples.append(', '.join(key_parts))

        return samples


3.3 差异分析服务（app/services/diff_service.py）

// python
import pandas as pd
import numpy as np
from typing import Dict, List, Any, Tuple
from app.models import ComparisonTask, ComparisonResult, File, db
from app.utils.key_detector import KeyDetector
import json

class DiffService:
    """差异分析服务"""

    @staticmethod
    def compare_files(task_id: int) -> Dict[str, Any]:
        """
        执行文件对比

        Args:
            task_id: 任务ID

        Returns:
            对比结果字典
        """
        # 获取任务信息
        task = ComparisonTask.query.get(task_id)
        if not task:
            raise ValueError(f"任务 {task_id} 不存在")

        # 加载文件
        file1 = File.query.get(task.file1_id)
        file2 = File.query.get(task.file2_id)

        if not file1 or not file2:
            raise ValueError("文件不存在")

        # 读取数据
        df1 = DiffService._load_excel(
            file1.file_path,
            task.sheet_name,
            task.header_row
        )
        df2 = DiffService._load_excel(
            file2.file_path,
            task.sheet_name,
            task.header_row
        )

        # 解析配置参数
        key_columns = json.loads(task.key_columns) if task.key_columns else []
        ignore_columns = json.loads(task.ignore_columns) if task.ignore_columns else []

        # 移除忽略列
        if ignore_columns:
            df1 = df1.drop(columns=[col for col in ignore_columns if col in df1.columns], errors='ignore')
            df2 = df2.drop(columns=[col for col in ignore_columns if col in df2.columns], errors='ignore')

        # 智能检测主键（如果未指定）
        key_detector = KeyDetector(df1, df2)
        if not key_columns:
            key_columns = key_detector.detect_key_columns()
            if key_columns:
                # 更新任务配置
                task.key_columns = json.dumps(key_columns)
                db.session.commit()

        # 构建主键索引
        if key_columns:
            key_index1 = key_detector.build_key_index(df1, key_columns)
            key_index2 = key_detector.build_key_index(df2, key_columns)
        else:
            # 无主键，使用行号作为标识
            key_index1 = {i: i for i in range(len(df1))}
            key_index2 = {i: i for i in range(len(df2))}

        # 执行对比
        result = DiffService._perform_comparison(
            df1, df2, key_index1, key_index2, key_columns
        )

        # 保存结果
        comparison_result = ComparisonResult(
            task_id=task.id,
            file1_row_count=len(df1),
            file2_row_count=len(df2),
            added_count=result['added_count'],
            deleted_count=result['deleted_count'],
            modified_count=result['modified_count'],
            unchanged_count=result['unchanged_count'],
            diff_details=json.dumps(result['diff_details'], ensure_ascii=False)
        )
        db.session.add(comparison_result)

        # 更新任务状态
        task.status = 'success'
        task.completed_at = pd.Timestamp.now()
        db.session.commit()

        return result

    @staticmethod
    def _load_excel(file_path: str, sheet_name: str, header_row: int) -> pd.DataFrame:
        """
        加载Excel文件

        Args:
            file_path: 文件路径
            sheet_name: Sheet名称
            header_row: 表头行号（0表示第一行）

        Returns:
            DataFrame
        """
        try:
            df = pd.read_excel(
                file_path,
                sheet_name=sheet_name,
                header=header_row
            )

            # 移除全空行
            df = df.dropna(how='all')

            # 标准化列名（去除前后空格）
            df.columns = df.columns.str.strip()

            return df
        except Exception as e:
            raise ValueError(f"读取Excel文件失败: {str(e)}")

    @staticmethod
    def _perform_comparison(
        df1: pd.DataFrame,
        df2: pd.DataFrame,
        key_index1: Dict[Tuple, int],
        key_index2: Dict[Tuple, int],
        key_columns: List[str]
    ) -> Dict[str, Any]:
        """
        执行具体对比逻辑

        Args:
            df1: 文件1数据
            df2: 文件2数据
            key_index1: 文件1的主键索引
            key_index2: 文件2的主键索引
            key_columns: 主键列名列表

        Returns:
            对比结果字典
        """
        all_keys = set(key_index1.keys()) | set(key_index2.keys())

        added_rows = []
        deleted_rows = []
        modified_rows = []
        unchanged_rows = []

        # 获取共同列（排除主键列）
        common_columns = list(set(df1.columns) & set(df2.columns))
        compare_columns = [col for col in common_columns if col not in key_columns]

        for key in all_keys:
            in_file1 = key in key_index1
            in_file2 = key in key_index2

            if in_file1 and not in_file2:
                # 删除的行
                row_idx = key_index1[key]
                row = df1.iloc[row_idx].to_dict()
                deleted_rows.append({
                    'key': str(key),
                    'row_number': row_idx + 1,
                    'data': row
                })

            elif not in_file1 and in_file2:
                # 新增的行
                row_idx = key_index2[key]
                row = df2.iloc[row_idx].to_dict()
                added_rows.append({
                    'key': str(key),
                    'row_number': row_idx + 1,
                    'data': row
                })

            else:
                # 检查修改
                row_idx1 = key_index1[key]
                row_idx2 = key_index2[key]

                row1 = df1.iloc[row_idx1]
                row2 = df2.iloc[row_idx2]

                # 对比每一列
                changes = []
                for col in compare_columns:
                    val1 = DiffService._normalize_value(row1.get(col))
                    val2 = DiffService._normalize_value(row2.get(col))

                    if val1 != val2:
                        changes.append({
                            'column': col,
                            'old_value': str(row1.get(col, '')),
                            'new_value': str(row2.get(col, ''))
                        })

                if changes:
                    modified_rows.append({
                        'key': str(key),
                        'row_number_file1': row_idx1 + 1,
                        'row_number_file2': row_idx2 + 1,
                        'changes': changes,
                        'data_file1': row1.to_dict(),
                        'data_file2': row2.to_dict()
                    })
                else:
                    unchanged_rows.append({
                        'key': str(key)
                    })

        return {
            'added_count': len(added_rows),
            'deleted_count': len(deleted_rows),
            'modified_count': len(modified_rows),
            'unchanged_count': len(unchanged_rows),
            'diff_details': {
                'added': added_rows,
                'deleted': deleted_rows,
                'modified': modified_rows,
                'unchanged': unchanged_rows,
                'key_columns': key_columns,
                'compare_columns': compare_columns
            }
        }

    @staticmethod
    def _normalize_value(value: Any) -> Optional[str]:
        """
        规范化值用于对比

        Args:
            value: 原始值

        Returns:
            规范化后的字符串值
        """
        if pd.isna(value) or value == '' or value is None:
            return None
        return str(value).strip()

// python
import pandas as pd
import os
from datetime import datetime
import uuid
from typing import Dict, Any
from openpyxl import Workbook
from openpyxl.styles import Font, PatternFill, Alignment, Border, Side
from openpyxl.utils.dataframe import dataframe_to_rows

class ReportService:
    """报告生成服务"""

    REPORTS_FOLDER = 'reports'

    # 样式定义
    HEADER_FILL = PatternFill(start_color='4472C4', end_color='4472C4', fill_type='solid')
    HEADER_FONT = Font(color='FFFFFF', bold=True, size=11)
    DATA_FONT = Font(size=10)

    ADDED_FILL = PatternFill(start_color='C6EFCE', end_color='C6EFCE', fill_type='solid')  # 绿色
    DELETED_FILL = PatternFill(start_color='FFC7CE', end_color='FFC7CE', fill_type='solid')  # 红色
    MODIFIED_FILL = PatternFill(start_color='FFEB9C', end_color='FFEB9C', fill_type='solid')  # 黄色

    @staticmethod
    def generate_report(task_id: int, diff_details: Dict[str, Any]) -> str:
        """
        生成Excel对比报告

        Args:
            task_id: 任务ID
            diff_details: 差异详情字典

        Returns:
            报告文件路径
        """
        # 确保报告目录存在
        os.makedirs(ReportService.REPORTS_FOLDER, exist_ok=True)

        # 生成唯一文件名
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        report_filename = f"diff_report_{timestamp}_{uuid.uuid4().hex[:8]}.xlsx"
        report_path = os.path.join(ReportService.REPORTS_FOLDER, report_filename)

        # 创建工作簿
        wb = Workbook()

        # 删除默认Sheet
        wb.remove(wb.active)

        # 生成各个Sheet
        ReportService._generate_summary_sheet(wb, diff_details)
        ReportService._generate_statistics_sheet(wb, diff_details)
        ReportService._generate_added_sheet(wb, diff_details['diff_details']['added'])
        ReportService._generate_deleted_sheet(wb, diff_details['diff_details']['deleted'])
        ReportService._generate_modified_sheet(wb, diff_details['diff_details']['modified'])

        # 保存文件
        wb.save(report_path)

        return report_path

    @staticmethod
    def _generate_summary_sheet(wb: Workbook, diff_details: Dict[str, Any]):
        """生成差异汇总Sheet"""
        ws = wb.create_sheet('差异汇总概览')

        # 写入标题
        ws['A1'] = 'APP测试报告对比分析报告'
        ws['A1'].font = Font(size=16, bold=True)
        ws.merge_cells('A1:E1')

        # 写入生成时间
        ws['A3'] = f'生成时间: {datetime.now().strftime("%Y-%m-%d %H:%M:%S")}'

        # 写入差异统计
        row = 5
        ws[f'A{row}'] = '统计项目'
        ws[f'B{row}'] = '数值'
        ReportService._apply_header_style(ws, row)

        stats = [
            ['文件1总行数', diff_details['file1_row_count']],
            ['文件2总行数', diff_details['file2_row_count']],
            ['新增测试项', diff_details['added_count']],
            ['删除测试项', diff_details['deleted_count']],
            ['修改测试项', diff_details['modified_count']],
            ['未改变测试项', diff_details['unchanged_count']],
            ['差异总计', diff_details['added_count'] + diff_details['deleted_count'] + diff_details['modified_count']]
        ]

        row += 1
        for key, value in stats:
            ws[f'A{row}'] = key
            ws[f'B{row}'] = value
            row += 1

        # 自动调整列宽
        ws.column_dimensions['A'].width = 20
        ws.column_dimensions['B'].width = 15

    @staticmethod
    def _generate_statistics_sheet(wb: Workbook, diff_details: Dict[str, Any]):
        """生成统计汇总Sheet"""
        ws = wb.create_sheet('详细统计')

        # 差异类型分布
        row = 1
        ws[f'A{row}'] = '差异类型'
        ws[f'B{row}'] = '数量'
        ws[f'C{row}'] = '占比'
        ReportService._apply_header_style(ws, row)

        total = diff_details['added_count'] + diff_details['deleted_count'] + diff_details['modified_count']

        types = [
            ('新增项', diff_details['added_count'], ReportService.ADDED_FILL),
            ('删除项', diff_details['deleted_count'], ReportService.DELETED_FILL),
            ('修改项', diff_details['modified_count'], ReportService.MODIFIED_FILL),
            ('未改变', diff_details['unchanged_count'], None)
        ]

        row += 1
        for name, count, fill in types:
            ws[f'A{row}'] = name
            ws[f'B{row}'] = count
            ws[f'C{row}'] = f'{count/total*100:.2f}%' if total > 0 else '0%'

            if fill:
                ws[f'A{row}'].fill = fill
                ws[f'B{row}'].fill = fill

            row += 1

        ws.column_dimensions['A'].width = 15
        ws.column_dimensions['B'].width = 12
        ws.column_dimensions['C'].width = 12

    @staticmethod
    def _generate_added_sheet(wb: Workbook, added_rows: list):
        """生成新增项详情Sheet"""
        ws = wb.create_sheet('新增项明细')

        if not added_rows:
            ws['A1'] = '无新增项'
            return

        # 写入表头
        headers = ['主键', '行号'] + list(added_rows[0]['data'].keys())
        for col, header in enumerate(headers, start=1):
            cell = ws.cell(row=1, column=col, value=header)
            cell.fill = ReportService.HEADER_FILL
            cell.font = ReportService.HEADER_FONT

        # 写入数据
        for row_idx, row_data in enumerate(added_rows, start=2):
            ws.cell(row=row_idx, column=1, value=row_data['key'])
            ws.cell(row=row_idx, column=2, value=row_data['row_number'])

            for col_idx, (key, value) in enumerate(row_data['data'].items(), start=3):
                ws.cell(row=row_idx, column=col_idx, value=value)
                ws.cell(row=row_idx, column=col_idx).fill = ReportService.ADDED_FILL

        # 自动调整列宽
        for col in range(1, len(headers) + 1):
            ws.column_dimensions[chr(64 + col)].width = 15

    @staticmethod
    def _generate_deleted_sheet(wb: Workbook, deleted_rows: list):
        """生成删除项详情Sheet"""
        ws = wb.create_sheet('删除项明细')

        if not deleted_rows:
            ws['A1'] = '无删除项'
            return

        # 写入表头
        headers = ['主键', '行号'] + list(deleted_rows[0]['data'].keys())
        for col, header in enumerate(headers, start=1):
            cell = ws.cell(row=1, column=col, value=header)
            cell.fill = ReportService.HEADER_FILL
            cell.font = ReportService.HEADER_FONT

        # 写入数据
        for row_idx, row_data in enumerate(deleted_rows, start=2):
            ws.cell(row=row_idx, column=1, value=row_data['key'])
            ws.cell(row=row_idx, column=2, value=row_data['row_number'])

            for col_idx, (key, value) in enumerate(row_data['data'].items(), start=3):
                ws.cell(row=row_idx, column=col_idx, value=value)
                ws.cell(row=row_idx, column=col_idx).fill = ReportService.DELETED_FILL

        # 自动调整列宽
        for col in range(1, len(headers) + 1):
            ws.column_dimensions[chr(64 + col)].width = 15

    @staticmethod
    def _generate_modified_sheet(wb: Workbook, modified_rows: list):
        """生成修改项详情Sheet"""
        ws = wb.create_sheet('修改项对比')

        if not modified_rows:
            ws['A1'] = '无修改项'
            return

        # 写入表头
        headers = ['主键', '行号(文件1)', '行号(文件2)', '变化列数', '变化详情']
        for col, header in enumerate(headers, start=1):
            cell = ws.cell(row=1, column=col, value=header)
            cell.fill = ReportService.HEADER_FILL
            cell.font = ReportService.HEADER_FONT

        # 写入数据
        for row_idx, row_data in enumerate(modified_rows, start=2):
            ws.cell(row=row_idx, column=1, value=row_data['key'])
            ws.cell(row=row_idx, column=2, value=row_data['row_number_file1'])
            ws.cell(row=row_idx, column=3, value=row_data['row_number_file2'])
            ws.cell(row=row_idx, column=4, value=len(row_data['changes']))

            # 生成变化详情字符串
            changes_text = '; '.join([
                f"{change['column']}: {change['old_value']} -> {change['new_value']}"
                for change in row_data['changes']
            ])
            ws.cell(row=row_idx, column=5, value=changes_text)

            # 高亮标记
            for col in range(1, 6):
                ws.cell(row=row_idx, column=col).fill = ReportService.MODIFIED_FILL

        ws.column_dimensions['A'].width = 20
        ws.column_dimensions['B'].width = 12
        ws.column_dimensions['C'].width = 12
        ws.column_dimensions['D'].width = 12
        ws.column_dimensions['E'].width = 80

    @staticmethod
    def _apply_header_style(ws, row: int):
        """应用表头样式"""
        for col in range(1, 6):
            cell = ws.cell(row=row, column=col)
            cell.fill = ReportService.HEADER_FILL
            cell.font = ReportService.HEADER_FONT
            cell.alignment = Alignment(horizontal='center')
